// FPSコントローラー const controls = new THREE.PointerLockControls(camera, document.body); // クリックでロック開始 document.body.addEventListener('click', () => { controls.lock(); }); // キーボード入力 const keys = {}; document.addEventListener('keydown', e => keys[e.code] = true); document.addEventListener('keyup', e => keys[e.code] = false); // 移動処理 function updateMovement() { const speed = 0.1; if (keys['KeyW']) controls.moveForward(speed); if (keys['KeyS']) controls.moveForward(-speed); if (keys['KeyA']) controls.moveRight(-speed); if (keys['KeyD']) controls.moveRight(speed); } updateMovement();
